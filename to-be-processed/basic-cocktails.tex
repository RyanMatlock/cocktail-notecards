\begin{Cocktail}{Caipirinha}
	\begin{Ingredients}
	\item \SI{1\half*}{\oz} cacha\c{c}a
	\item half a lime, cut into 3 or 4 wedges
	\item \SI{2}{\tsp} sugar
	\end{Ingredients}
	
	\begin{Instructions}
	Muddle lime and sugar together in an old fashioned glass, fill the glass with ice, and add the cacha\c{c}a, and you've made Brazil's national cocktail.
	
	Cacha\c{c}a, like rum, is fermented from sugar cane, so in a pinch, rum can be subtituted for cacha\c{c}a.
	\end{Instructions}
\end{Cocktail}

\begin{Cocktail}{Old Fashioned}
	\begin{Ingredients}
	\item \SI{1\half*}{\oz} whiskey
	\item a few \si{\dashes} Angostura bitters
	\item 1 sugar cube
	\item a \si{\splash} \soda
	\end{Ingredients}
	
	\begin{Instructions}\small
	Saturate sugar cube with bitters in an old fashioned glass, add a splash of \soda, and muddle the sugar until dissolved.  Fill the glass with ice, add the whiskey, and garnish with an orange slice, a lemon twist, and two maraschino cherries.
	
	Bourbon is the most commonly used type of whiskey, although rye whiskey is also acceptable.  The \InlineCocktail{Brandy Old Fahioned} was once a popular variation---simply substitute brandy for whiskey.
	\end{Instructions}
\end{Cocktail}
